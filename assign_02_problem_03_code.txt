public class RightAngleTriangleProgram {
  public static void main(String[] args) {
    int base = 5;

    for (int i = 1; i <= base; i++) {
      for (int j = 1; j <= i; j++) {
        System.out.print("*");
      }
      System.out.println();
    }
  }
}
